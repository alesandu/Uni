			;	 }
			; MCD(A,B) = A
			
varA			EQU		75
varB			EQU		33
MCD			DCD		1
			
			MOV		R1, #varA	; R1 <- A (indirizzamento INDIRETTO)
			MOV		R2, #varB	; R2 <- B (indirizzamento INDIRETTO)
			
Loop
			CMP		R1,R2		; Confronto A e B
			BGT		a_maggiore_b	; se A > B
			BLT		a_minore_b	; se A < B 
			BEQ		a_uguale_b	; se A == B
			
a_maggiore_b
			SUB		R1, R1, R2	; A = A - B
			BAL		Loop
			
a_minore_b
			SUB		R2, R2, R1	; B = B - A
			BAL		Loop
			
a_uguale_b
			MOV		R0, R1		; R0 <- A
			LDR		R4, =MCD	; R4 <- indirizzo di MCD
			STR		R0, [R4]	; indirizzamento indiretto con Registro ( PUNTATORE )
