		;		MOLTIPLICAZIONE PER ADDIZIONI SUCCESSIVE
		;
		;		int risultato = 0;
		;		int x = 9;
		;		int y = 9;
		;
		;		int i = 0;
		;
		;		while( i < y ) {
		;		risultato = risultato + x;
		;		i ++ ;
		;		}
		
		
varX		EQU		7
varY		EQU		8

		; indirizzamento immediato
		MOV		R0, #0	; R0 <- 0 ( risultato = 0 )
		MOV		R1, #0  ; R1 <- 0 ( indice i = 0 )


Ciclo
		; op1 indirizzamento a registro, op2 indirizzamento indiretto
		ADD		R0, R0, #varX		; risultato = risultato + x
		; op1 indirizzamento a registro, op2 indirizzamento immediato
		ADD		R1, R1, #1		; i ++
		CMP		R1, #varY		; confronto se i < y
		BLT		Ciclo			; if conditio = TRUE GOTO Ciclo

		END
